# vagrant

# BOXの取得
vagrant box add [BOX名] [URL]

# BOXリスト
vagrant box list

# BOXを適用
vagrant init [BOX名]

# 仮装環境を起動
[vagrantfileがあるディレクトリで] vagrant up

# 以下、起動中の各種コマンド
vagrant status
vagrant suspend 一旦停止
vagrant resume 復帰
vagrant halt 終了
vagrant reload 再起動
vagrant destroy 仮想マシンの削除

# ssh接続
vagrant ssh
